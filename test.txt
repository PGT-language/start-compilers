```intro2.pgt
package main

function(main) {
    filename + string = "demo.txt"
    content + string = "Hello World !"
    
    println("=== File Operations Demo ===")
    
    println("1. Creating file...")
    create::file(filename)
    
    println("2. Writing to file...")
    write::file(filename, content)
    write::file(filename, "Second line")
    
    
    println("=== All operations completed! ===")
    return 1
}

return 0

```

    println(" = ", diff, "{int}")
    
    println(a, "{int}")
    println(" * ")
    println(b, "{int}")
    println(" = ", prod, "{int}")
    
    println(a, "{int}")
